def performance_metric2(y_true, y_predict):
    """"""
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    N = y_true.size
    yBar = y_true.sum()/ N
    SSR = ((y_true - y_predict) ** 2).sum()
    SST = ((y_true - yBar) ** 2).sum()
    score = 1 - SSR / SST

    return score
